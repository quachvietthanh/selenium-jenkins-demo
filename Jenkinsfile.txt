pipeline {
    agent any
    environment {
        CHROME_DRIVER_PATH = "/usr/local/bin/chromedriver"
    }
    stages {
        stage('Checkout Code') {
            steps {
                git branch: 'main', url: 'https://github.com/your-username/selenium-test-project.git'
            }
        }
        stage('Build') {
            steps {
                sh 'mvn clean compile'
            }
        }
        stage('Run Selenium Tests') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Post-Test Report') {
            steps {
                junit 'target/surefire-reports/*.xml'
            }
        }
    }
}
